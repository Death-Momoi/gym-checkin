function processPreCheckOut() { var name = localStorage.getItem('gym_username'); showLoading("正在驗證身分權限..."); // 💡 關鍵改動：簽退前，先發送 POST 去後端查查看自己有沒有被轉移成「主要借用者」 fetch(GAS_URL, { method: "POST", body: JSON.stringify({ action: "檢查目前身分", name: name }) }) .then(res => res.json()) .then(response => { hideLoading(); var latestRole = response.role || '共同使用者'; localStorage.setItem('gym_current_role', latestRole); // 同步更新本地端的身分記錄 if (latestRole === '主要借用者') { // 成功捕捉到「被指定者」的身分轉移！強制進入檢查與轉移畫面 showLoading("正在獲取現場人員名單..."); fetch(GAS_URL, { method: "POST", body: JSON.stringify({ action: "取得在場名單", excludeName: name }) }) .then(res => res.json()) .then(response => { hideLoading(); var select = document.getElementById('select-next-user'); select.innerHTML = ''; if (response.users && response.users.length > 0) { response.users.forEach(function(username) { var opt = document.createElement('option'); opt.value = username; opt.innerText = username; select.appendChild(opt); }); } document.getElementById('step-main').classList.add('hidden'); document.getElementById('step-device').classList.remove('hidden'); }); } else { // 依舊是共同使用者，直接送出簽退 var bubble = document.getElementById('btn-bubble'); bubble.innerText = "處理中..."; var payload = { action: '簽退', name: name, email: localStorage.getItem('gym_email'), role: '共同使用者' }; fetch(GAS_URL, { method: "POST", body: JSON.stringify(payload) }).then(() => { bubble.innerText = "簽退成功"; localStorage.setItem('gym_status', 'checked_out'); setTimeout(updateBubbleUI, 1500); }); } }) .catch(err => { alert("身分驗證失敗，請重新再試"); hideLoading(); }); }
